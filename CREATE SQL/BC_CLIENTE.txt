CREATE TABLE bc_cliente (
    ID_CLIENTE VARCHAR(40) NOT NULL PRIMARY KEY,
    CODIGO_CLIENTE VARCHAR(6) NOT NULL,
    USUARIO VARCHAR(20) NOT NULL,
    NOME VARCHAR(70) NOT NULL,
    LOJA_CLIENTE varchar(2) NOT NULL	
    TIPO VARCHAR(1) NOT NULL,
    CNPJ_CPF VARCHAR(18) NOT NULL,
    IE VARCHAR(18),
    DT_NASCIMENTO DATETIME NOT NULL,
    TELEFONE VARCHAR(15),
    EMAIL VARCHAR(100),
    CEP VARCHAR(8) NOT NULL,
    ENDERECO VARCHAR(80) NOT NULL,
    BAIRRO VARCHAR(70) NOT NULL,
    MUNICIPIO VARCHAR(50) NOT NULL,
    CODIGO_MUNICIPIO VARCHAR(7) NOT NULL,
    ESTADO VARCHAR(2) NOT NULL,
    LIMITE_CREDITO DECIMAL(14,2) NOT NULL,
    BLOQUEADO VARCHAR(1) NOT NULL,
    QTDCOMPRAS INT NOT NULL,
    TOTALCOMPRAS DECIMAL(14,2) NOT NULL,
    QTDDEVOLUCOES INT NOT NULL,
    TOTALDEVOLUCOES DECIMAL(14,2) NOT NULL,
    USR_ALT VARCHAR(6),
    DT_ALT TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    USR_CAD VARCHAR(6),
    DT_CAD TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PEND BOOLEAN NOT NULL,
    LOG TEXT
);
