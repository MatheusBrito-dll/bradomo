CREATE TABLE `brd`.`or_usuario` (
    `ID_USUARIO` VARCHAR(40) NOT NULL,
    `CODIGO_USUARIO` VARCHAR(6) NOT NULL,
    `USUARIO` VARCHAR(20) NOT NULL,
    `NOME` VARCHAR(20) NOT NULL,
    `SOBRENOME` VARCHAR(30) NOT NULL,
    `EMAIL` VARCHAR(50) NOT NULL,
    `CPF` CHAR(11) NOT NULL,
    `ENDERECO` TEXT NOT NULL,
    `SENHA` VARCHAR(255) NOT NULL, -- Ajuste conforme o m√©todo de hashing usado
    `NASCIMENTO` DATE NOT NULL,
    `BLOQ` BOOLEAN NOT NULL,
    `CARGO` VARCHAR(10) NOT NULL,
    `NIVEL` INT NOT NULL,
    `USR_ALT` VARCHAR(6),
    `DT_ALT` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    `USR_CAD` VARCHAR(6),
    `DT_CAD` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    `PEND` BOOLEAN NOT NULL,
    `LOG` TEXT,
    PRIMARY KEY (`ID_USUARIO`),
    UNIQUE (`CODIGO_USUARIO`)
) ENGINE = InnoDB;
