CREATE TABLE brd_master.rs_mesa (
    `ID_MESA` VARCHAR(40) NOT NULL,
    `NUMERO` INT NOT NULL, 
    `ID_CLIENTE` VARCHAR(40) NOT NULL,
    `HORARIO_RESERVA` TIMESTAMP,
    `HORARIO_CHEGADA` TIMESTAMP,
    `STATUS`INT NOT NULL,
    `ID_PEDIDO` INT,
    `OBSERVACAO` TEXT,
    `USR_ALT` VARCHAR(6),
    `DT_ALT` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    `USR_CAD` VARCHAR(6),
    `DT_CAD` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    `PEND` BOOLEAN NOT NULL,
    `LOG` TEXT,
    FOREIGN KEY (ID_CLIENTE) REFERENCES bc_cliente(ID_CLIENTE),
    FOREIGN KEY (ID_PEDIDO) REFERENCES sl_pedido(ID_PEDIDO)
);
